## 344. 反转字符串
定义应该`temp`
```cpp
class Solution {
public:
    void reverseString(vector<char>& s) {
        int left = 0, right = s.size()-1;
        while(left < right){
            char temp = s[left];
            s[left] = s[right];
            s[right] = temp;
            left++;
            right--;
        }
    }
};
```

```cpp
	class Solution {
	public:
	    void reverseString(vector<char>& s) {
	        for (int i = 0, j = s.size() - 1; i < s.size()/2; i++, j--) {
	            swap(s[i],s[j]);
	        }
	    }
	};
```
